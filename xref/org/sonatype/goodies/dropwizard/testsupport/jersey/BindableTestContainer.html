13  package org.sonatype.goodies.dropwizard.testsupport.jersey;
14  
15  import java.net.URI;
16  
17  import javax.ws.rs.core.UriBuilder;
18  
19  import org.glassfish.hk2.utilities.Binder;
20  import org.glassfish.jersey.client.ClientConfig;
21  import org.glassfish.jersey.server.ApplicationHandler;
22  import org.glassfish.jersey.test.DeploymentContext;
23  import org.glassfish.jersey.test.inmemory.ExposedInMemoryConnectorProvider;
24  import org.glassfish.jersey.test.spi.TestContainer;
25  import org.glassfish.jersey.test.spi.TestContainerFactory;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import static com.google.common.base.Preconditions.checkNotNull;
30  
31  /**
32   * {@link TestContainer} which exposes {@link Binder}.
33   *
34   * @since ???
35   */
36  public class BindableTestContainer
37      implements TestContainer
38  {
39    private static final Logger log = LoggerFactory.getLogger(BindableTestContainer.class);
40  
41    private final URI baseUri;
42  
43    private final ApplicationHandler handler;
44  
45    public BindableTestContainer(final URI baseUri, final DeploymentContext context, final Binder binder) {
46      checkNotNull(baseUri);
47      checkNotNull(context);
48      checkNotNull(binder);
49      this.baseUri = UriBuilder.fromUri(baseUri).path(context.getContextPath()).build();
50      this.handler = new ApplicationHandler(context.getResourceConfig(), binder);
51    }
52  
53    @Override
54    public URI getBaseUri() {
55      return baseUri;
56    }
57  
58    public ApplicationHandler getHandler() {
59      return handler;
60    }
61  
62    @Override
63    public ClientConfig getClientConfig() {
64      return new ClientConfig().connectorProvider(new ExposedInMemoryConnectorProvider(baseUri, handler));
65    }
66  
67    @Override
68    public void start() {
69      log.debug("Started");
70    }
71  
72    @Override
73    public void stop() {
74      log.debug("Stopped");
75    }
76  
77    public static class Factory
78        implements TestContainerFactory
79    {
80      private final Binder binder;
81  
82      public Factory(final Binder binder) {
83        this.binder = checkNotNull(binder);
84      }
85  
86      @Override
87      public TestContainer create(final URI baseUri, final DeploymentContext context) {
88        return new BindableTestContainer(baseUri, context, binder);
89      }
90    }
91  }
