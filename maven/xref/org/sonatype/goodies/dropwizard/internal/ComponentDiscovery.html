43  /**
44   * Component discovery.
45   *
46   * @since 1.0.0
47   */
48  @Named
49  @Singleton
50  public class ComponentDiscovery
51  {
52    private static final Logger log = LoggerFactory.getLogger(ComponentDiscovery.class);
53  
54    private final BeanLocator locator;
55  
56    @Inject
57    public ComponentDiscovery(final BeanLocator locator) {
58      this.locator = checkNotNull(locator);
59    }
60  
61    /**
62     * Discover components.
63     */
64    public void discover(final Environment environment) {
65      log.debug("Discovering components");
66  
67      detectHealthChecks(environment);
68      detectJaxrsComponents(environment);
69      detectTasks(environment);
70      detectManaged(environment);
71      detectCustomizers(environment);
72    }
73  
74    /**
75     * Locate components for given type with optional specific qualifier annotation.
76     */
77    @SuppressWarnings("unchecked")
78    private <Q extends Annotation, C> Iterable<BeanEntry<Q, C>> locate(final Class<C> type, @Nullable final Class<Q> annotation) {
79      List<BeanEntry<Q, C>> result = new ArrayList<>();
80  
81      Key<C> key;
82      if (annotation != null) {
83        key = Key.get(type, annotation);
84      }
85      else {
86        key = Key.get(type);
87      }
88  
89      for (BeanEntry entry : locator.locate(key)) {
90        result.add(entry);
91      }
92  
93      return result;
94    }
95  
96    /**
97     * Locate components for given type.
98     */
99    private <C> Iterable<BeanEntry<Annotation, C>> locate(final Class<C> type) {
100     return locate(type, null);
101   }
102 
103   // TODO: may need a better way to order managed dependencies which are started in order they are added

106   /**
107    * Detect {@link Managed} components.
108    */
109   private void detectManaged(final Environment environment) {
110     for (BeanEntry<Annotation, Managed> entry : locate(Managed.class)) {
111       Managed component = entry.getValue();
112       environment.lifecycle().manage(component);
113       log.debug("Added managed: {}", component);
114     }
115   }
116 
117   /**
118    * Detect {@link Task} components.
119    */
120   private void detectTasks(final Environment environment) {
121     for (BeanEntry<Annotation, Task> entry : locate(Task.class)) {
122       Task component = entry.getValue();
123       environment.admin().addTask(component);
124       log.debug("Added task: {}", component);
125     }
126   }
127 
128   /**
129    * Detect {@link HealthCheck} components.
130    */
131   private void detectHealthChecks(final Environment environment) {
132     for (BeanEntry<Named, HealthCheck> entry : locate(HealthCheck.class, Named.class)) {
133       String name = entry.getKey().value();
134       HealthCheck component = entry.getValue();
135       environment.healthChecks().register(name, component);
136       log.debug("Added health-check: {} -> {}", name, component);
137     }
138   }
139 
140   /**
141    * Detect JAX-RS components.
142    */
143   private void detectJaxrsComponents(final Environment environment) {
144     // can not use @Provider here, as its not a Guice @Qualifier annotation
145     for (BeanEntry<Annotation, Component> entry : locate(Component.class)) {
146       Component component = entry.getValue();
147       environment.jersey().register(component);
148       log.debug("Added JAX-RS component: {}", component);
149     }
150 
151     for (BeanEntry<Annotation, Resource> entry : locate(Resource.class)) {
152       Resource component = entry.getValue();
153       environment.jersey().register(component);
154       log.debug("Added JAX-RS resource: {}", component);
155     }
156 
157     for (BeanEntry<Annotation, ExceptionMapper> entry : locate(ExceptionMapper.class)) {
158       ExceptionMapper component = entry.getValue();
159       environment.jersey().register(component);
160       log.debug("Added JAX-RS exception-mapper: {}", component);
161     }
162 
163     for (BeanEntry<Annotation, Feature> entry : locate(Feature.class)) {
164       Feature component = entry.getValue();
165       environment.jersey().register(component);
166       log.debug("Added JAX-RS feature: {}", component);
167     }
168 
169     for (BeanEntry<Annotation, DynamicFeature> entry : locate(DynamicFeature.class)) {
170       DynamicFeature component = entry.getValue();
171       environment.jersey().register(component);
172       log.debug("Added JAX-RS dynamic-feature: {}", component);
173     }
174   }
175 
176   /**
177    * Detect and apply {@link EnvironmentCustomizer} components.
178    */
179   private void detectCustomizers(final Environment environment) {
180     for (BeanEntry<Annotation, EnvironmentCustomizer> entry : locate(EnvironmentCustomizer.class)) {
181       EnvironmentCustomizer customizer = entry.getValue();
182       log.debug("Applying customizer: {}", customizer);
183       customizer.customize(environment);
184     }
185   }
186 }
